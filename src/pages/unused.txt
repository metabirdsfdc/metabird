import { useCallback, useEffect, useMemo, useState } from "react";
import { useOrganizationMappingStore } from "./useOrganizationMapping";

/* ----------------------------------
 * Core Types
 * ---------------------------------- */

export type MetadataComponent = {
  id: string;
  name: string;
  componentType: string;
  category: "Metadata" | "Code";
  lastUpdated: string;
  isAvailable: boolean;
};

export type RetrieveSummary = {
  status: "IDLE" | "LOADING" | "SUCCESS" | "FAILED";
  message?: string;
};

/* ----------------------------------
 * Hook Contract
 * ---------------------------------- */

type UseRetrieveResult = {
  sourceOrg: string | null;

  componentType: string;
  category: string;

  listOfComponents: MetadataComponent[];
  filteredComponents: MetadataComponent[];
  selectedComponents: MetadataComponent[];

  retrieveComponents: () => Promise<void>;
  toggleComponentSelection: (id: string) => void;
  clearSelection: () => void;

  setComponentType: (value: string) => void;
  setCategory: (value: string) => void;

  isLoading: boolean;
  errorMessage: string | null;
  summary: RetrieveSummary;
};

/* ----------------------------------
 * Hook Implementation
 * ---------------------------------- */

export const useRetrieve = (): UseRetrieveResult => {
  const { sourceOrg } = useOrganizationMappingStore();

  const [componentType, setComponentType] = useState("");
  const [category, setCategory] = useState("");

  const [listOfComponents, setListOfComponents] = useState<MetadataComponent[]>(
    []
  );

  const [selectedComponents, setSelectedComponents] = useState<
    MetadataComponent[]
  >([]);

  const [isLoading, setIsLoading] = useState(false);
  const [errorMessage, setErrorMessage] = useState<string | null>(null);

  const [summary, setSummary] = useState<RetrieveSummary>({
    status: "IDLE"
  });

  /* ----------------------------------
   * Fetch Components (based on sourceOrg)
   * ---------------------------------- */

  const retrieveComponents = useCallback(async () => {
    if (!sourceOrg) {
      setErrorMessage("Source organization is not selected");
      return;
    }

    setIsLoading(true);
    setErrorMessage(null);
    setSummary({ status: "LOADING" });

    try {
      await new Promise((r) => setTimeout(r, 1200));

      const mockComponents: MetadataComponent[] = [
        {
          id: "1",
          name: "AccountController",
          componentType: "ApexClass",
          category: "Code",
          lastUpdated: "2025-12-10",
          isAvailable: true
        },
        {
          id: "2",
          name: "ContactTrigger",
          componentType: "ApexTrigger",
          category: "Code",
          lastUpdated: "2025-12-08",
          isAvailable: false
        },
        {
          id: "3",
          name: "Account",
          componentType: "CustomObject",
          category: "Metadata",
          lastUpdated: "2025-11-29",
          isAvailable: true
        }
      ];

      setListOfComponents(mockComponents);
      setSummary({
        status: "SUCCESS",
        message: `Retrieved ${mockComponents.length} components from ${sourceOrgId}`
      });
    } catch {
      setSummary({ status: "FAILED", message: "Retrieve failed" });
      setErrorMessage("Failed to retrieve components");
    } finally {
      setIsLoading(false);
    }
  }, [sourceOrg]);

  /* ----------------------------------
   * Filtering Logic
   * ---------------------------------- */

  const filteredComponents = useMemo(() => {
    return listOfComponents.filter((c) => {
      if (componentType && c.componentType !== componentType) return false;
      if (category && c.category !== category) return false;
      return true;
    });
  }, [listOfComponents, componentType, category]);

  /* ----------------------------------
   * Selection Logic
   * ---------------------------------- */

  const toggleComponentSelection = (id: string) => {
    setSelectedComponents((prev) => {
      const exists = prev.find((c) => c.id === id);
      if (exists) {
        return prev.filter((c) => c.id !== id);
      }
      const component = listOfComponents.find((c) => c.id === id);
      return component ? [...prev, component] : prev;
    });
  };

  const clearSelection = () => setSelectedComponents([]);

  /* ----------------------------------
   * Auto-reset when sourceOrg changes
   * ---------------------------------- */

  useEffect(() => {
    setListOfComponents([]);
    setSelectedComponents([]);
    setComponentType("");
    setCategory("");
    setSummary({ status: "IDLE" });
  }, [sourceOrg]);

  return {
    sourceOrg,

    componentType,
    category,

    listOfComponents,
    filteredComponents,
    selectedComponents,

    retrieveComponents,
    toggleComponentSelection,
    clearSelection,

    setComponentType,
    setCategory,

    isLoading,
    errorMessage,
    summary
  };
};
